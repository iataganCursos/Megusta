replace
$frase = "O rato roeu a roupa do rei de Roma.";
$nova_frase = str_replace("rato", "gato", $frase);
echo $nova_frase;

 

length
$minhaString = "Isso é uma string de exemplo.";
$comprimento = strlen($minhaString);
echo "O comprimento da string é: $comprimento";

 

substring
$minhaString = "Esta é uma string de exemplo";
$subString = substr($minhaString, 5, 6);
echo $subString;

 

charAt

$minhaString = "Exemplo de string";
$primeiroCaractere = $minhaString[0]; // Acessa o primeiro caractere (índice 0)
echo $primeiroCaractere; // Isso irá imprimir "E"

$terceiroCaractere = $minhaString[2]; // Acessa o terceiro caractere (índice 2)
echo $terceiroCaractere; // Isso irá imprimir "e"

 

indexOf
$minhaString = "Esta é uma string de exemplo";
$posicao = strpos($minhaString, "é");
echo "A posição da substring 'é' na string é: $posicao";

 

lastIndexOf
$minhaString = "Esta é uma string de exemplo com uma palavra exemplo.";
$posicao = strrpos($minhaString, "exemplo");
echo "A posição da última ocorrência de 'exemplo' na string é: $posicao";

 

toLowerCase
$minhaString = "Isso é uma String de Exemplo";
$string_em_minusculas = strtolower($minhaString);
echo $string_em_minusculas;

 

toUpperCase
$minhaString = "Isso é uma string de exemplo";
$string_em_maiusculas = strtoupper($minhaString);
echo $string_em_maiusculas;

 

compareTo

$string1 = "Hello";
$string2 = "hello";

$resultado = strcmp($string1, $string2);

if ($resultado === 0) {
echo "As strings são iguais.";
} elseif ($resultado < 0) {
echo "A primeira string é menor.";
} else {
echo "A primeira string é maior.";
}

 

compareToIgnoreCase

$string1 = "Hello";
$string2 = "hello";

$resultado = strcasecmp($string1, $string2);

if ($resultado === 0) {
echo "As strings são iguais (ignorando a diferença entre maiúsculas e minúsculas).";
} else {
echo "As strings são diferentes (ignorando a diferença entre maiúsculas e minúsculas).";
}

 

equals

$string1 = "Hello";
$string2 = "hello";

if ($string1 === $string2) {
echo "As strings são idênticas.";
} else {
echo "As strings são diferentes.";
}
