Array em Python

add

# Criando uma lista vazia
my_list = []

# Adicionando elementos à lista usando append()
my_list.append(1)
my_list.append(2)
my_list.append(3)

# A lista agora contém [1, 2, 3]
print(my_list)

 

size

# Usando len para obter o número de elementos em uma lista
my_list = [1, 2, 3, 4, 5, 6, 7, 8, 9]
list_size = len(my_list)

# Imprimindo o tamanho da lista
print(list_size)

set

remove

my_list = [10, 20, 30, 40, 50]

# Removendo o item na posição 2 (índice 1) usando pop()
removed_item = my_list.pop(1)

# Imprimindo a lista após a remoção
print(my_list)

# Imprimindo o item removido
print("Item removido:", removed_item)

 

clear

lastIndexOf

my_list = [10, 20, 30, 40, 30, 50]

# Invertendo a lista e encontrando o índice do elemento 30
reversed_list = my_list[::-1]
last_index = len(my_list) - reversed_list.index(30) - 1

# Imprimindo o último índice do elemento 30
print("O último índice do elemento 30 é:", last_index)
