Math.sign
def sign(numero):
    if numero > 0:
        return 1
    elif numero < 0:
        return -1
    else:
        return 0

# Exemplos de uso
print(sign(5))    # Saída: 1
print(sign(-3.14)) # Saída: -1
print(sign(0))     # Saída: 0
