substring

minha_string = "Python é uma linguagem de programação"

# Extrair os primeiros 6 caracteres
substring1 = minha_string[0:6] # Equivalente a minha_string[:6]
print(substring1) # Isso irá imprimir "Python"

# Extrair os caracteres de 7 a 18
substring2 = minha_string[7:19] # Equivalente a minha_string[7:19]
print(substring2) # Isso irá imprimir "é uma lingua"

# Extrair os últimos 6 caracteres
substring3 = minha_string[-6:] # Equivalente a minha_string[25:]
print(substring3) # Isso irá imprimir "ção"

indexOf
minha_string = "Python é uma linguagem de programação"
indice = minha_string.find("é")
if indice != -1:
print(f"A substring 'é' foi encontrada no índice {indice}")
else:
print("A substring não foi encontrada")

 

lastIndexOf

minha_string = "Python é uma linguagem de programação. Python é muito popular."

# Inverter a string
string_invertida = minha_string[::-1]

# Encontrar a última ocorrência de "Python" na string invertida
indice_invertido = string_invertida.rfind("Python")

# Calcular o índice na string original
indice_original = len(minha_string) - indice_invertido - 1

if indice_invertido != -1:
print(f"A última ocorrência de 'Python' começa no índice {indice_original}")
else:
print("A substring não foi encontrada")

compareTo

string1 = "python"
string2 = "Python"

# Comparar se duas strings são iguais (sem diferenciação entre maiúsculas e minúsculas)
if string1.lower() == string2.lower():
print("As strings são iguais (ignorando maiúsculas e minúsculas)")

# Comparar se uma string é maior que a outra (ordem alfabética)
if string1 > string2:
print("string1 vem depois de string2 no dicionário")

# Comparar se uma string é menor que a outra (ordem alfabética)
if string1 < string2:
print("string1 vem antes de string2 no dicionário")
