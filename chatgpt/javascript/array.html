Array em Javascript

add
const minhaArray = [1, 2, 3];
minhaArray.push(4);
// Agora, minhaArray é [1, 2, 3, 4]

 

size

const minhaArray = [1, 2, 3, 4, 5];
const tamanhoDaArray = minhaArray.length;

console.log(tamanhoDaArray); // Saída: 5

 

get

const minhaArray = [10, 20, 30, 40, 50];

// Acessando o elemento no índice 2
const elementoNoIndice2 = minhaArray[2];

console.log(elementoNoIndice2); // Saída: 30

 

set

const minhaArray = [10, 20, 30, 40, 50];

// Modificando o elemento no índice 2
minhaArray[2] = 35;

console.log(minhaArray); // Saída: [10, 20, 35, 40, 50]

 

remove

const minhaArray = [1, 2, 3, 4, 5];

// Remover um elemento a partir do índice 2
minhaArray.splice(2, 1);

console.log(minhaArray); // Saída: [1, 2, 4, 5]

 

clear

let minhaArray = [1, 2, 3, 4, 5];

// Esvaziar a array
minhaArray = [];

console.log(minhaArray); // Saída: []

 

contains

const minhaArray = [1, 2, 3, 4, 5];
const elemento = 3;

const contemElemento = minhaArray.includes(elemento);

console.log(contemElemento); // Saída: true

 

toArray

const meuObjeto = { chave1: 'valor1', chave2: 'valor2', chave3: 'valor3' };
const minhaArray = Object.values(meuObjeto);

console.log(minhaArray); // Saída: ['valor1', 'valor2', 'valor3']

 

indexOf

const minhaArray = [10, 20, 30, 40, 50];

const indiceDoElemento30 = minhaArray.indexOf(30);
console.log(indiceDoElemento30); // Saída: 2

const indiceDoElemento60 = minhaArray.indexOf(60);
console.log(indiceDoElemento60); // Saída: -1

 

lastIndexOf 

const minhaArray = [10, 20, 30, 40, 30, 50];

const ultimoIndiceDoElemento30 = minhaArray.lastIndexOf(30);
console.log(ultimoIndiceDoElemento30); // Saída: 4

const ultimoIndiceDoElemento60 = minhaArray.lastIndexOf(60);
console.log(ultimoIndiceDoElemento60); // Saída: -1
