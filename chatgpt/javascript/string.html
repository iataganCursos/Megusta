compareToIgnoreCase

function compareToIgnoreCase(string1, string2) {
const str1 = string1.toLowerCase();
const str2 = string2.toLowerCase();

if (str1 < str2) {
return -1;
} else if (str1 > str2) {
return 1;
} else {
return 0;
}
}

const resultado = compareToIgnoreCase("Maçã", "maÇã");
console.log(resultado); // Saída: 0 (strings são consideradas iguais, independentemente das maiúsculas/minúsculas)

equalsIgnoreCase

function equalsIgnoreCase(string1, string2) {
const str1 = string1.toLowerCase();
const str2 = string2.toLowerCase();

return str1 === str2;
}

const saoIguais = equalsIgnoreCase("Maçã", "maÇã");
console.log(saoIguais); // Saída: true (as strings são consideradas iguais, independentemente das maiúsculas/minúsculas)
